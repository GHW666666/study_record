## 移动端封装公共弹窗解决兼容性问题

问题点：
1. 键盘弹出调的是端协议hybrid，这是异步的协议，首先inputRef.value?.focus();需要等待端看到键盘处理完成返回键盘高度。如果直接在inputRef.value?.focus();之后await getKeyboardHeight，端可能还没有处理完，导致键盘高度为0。
getKeyBoardHeight 调的是端侧协议，而键盘弹起是异步的。
调 focus 后键盘不会瞬间弹出，需要动画时间（通常 200-400ms）。如果 focus 后立即调一次 getKeyBoardHeight，端侧可能返回 0（键盘还没弹出来）。
2. 事件的处理。
## @click.self="handleClose" 蒙层
阻止弹窗点击事件的冒泡，防止点击弹窗内部触发关闭事件。

## @touchmove.prevent 弹窗（默认事件）
阻止蒙层触摸的默认移动事件，防止mask蒙层触发默认的触摸移动事件。

![alt text](image-3.png)

## @touchmove.stop.prevent 弹窗
阻止弹窗触摸移动事件的冒泡，防止弹窗外部触发触摸移动事件。

## @touchstart.prevent="handleConfirm" 确认按钮

InputDialog 当初加 @touchstart.prevent 是为了解决"置灰按钮点击导致键盘收起"的问题

![alt text](image-4.png)


<template>
    <div
        v-if="visible"
        class="dialog-mask"
        :class="{ 'is-ready': ready }"
        @click.self="handleClose"
        @touchmove.prevent
    >
        <div
            class="dialog-panel"
            @touchmove.stop.prevent
        >
            <!-- 顶部栏 -->
            <div class="dialog-header">
                <div
                    class="header-cancel"
                    @click="handleClose"
                >
                    {{ cancelText }}
                </div>
                <div class="header-title">{{ title }}</div>
                <div
                    class="header-confirm"
                    :class="{ disabled: !canSubmit }"
                    @touchstart.prevent="handleConfirm"
                    @click="handleConfirm"
                >
                    {{ confirmText }}
                </div>
            </div>

            <!-- 输入框 -->
            <div class="form">
                <input
                    ref="inputRef"
                    v-model="inputValue"
                    class="name-input"
                    type="text"
                    :placeholder="placeholder"
                    :maxlength="maxlength"
                    @focus="handleInputFocus"
                />
            </div>

            <!-- 底部提示区 -->
            <div class="form-footer">
                <div
                    class="error-tip"
                    :class="{ 'is-visible': errorTip || showEmojiError }"
                >
                    {{ showEmojiError ? '暂不支持输入表情' : (errorTip || '占位') }}
                </div>
                <div class="char-count">{{ inputValue.length }}/{{ maxlength }}</div>
            </div>
        </div>
    </div>
</template>

<script lang="ts" setup>
import { ref, computed, watch, nextTick, onMounted } from 'vue';
import { getKeyBoardHeight, isIos } from '@share/composables/hybridExecute.ts';
import { registerSetKeyboardHeight } from '@share/composables/hybridRegister.ts';

const kbHeight = ref(0);
const kbHeightCache = ref(0);
const panelBottom = computed(() => {
    if (isIos) return '0px';
    return `${kbHeight.value}px`;
});

const fetchKeyboardHeight = async () => {
    for (let i = 0; i < 5; i++) { // 最多轮询 ~1.5s
        const res = await getKeyBoardHeight();
        if (!(res instanceof Error) && res?.keyboardHeight > 50) {
            kbHeight.value = Number(res?.keyboardHeight) || 0;
            return;
        }
        await new Promise(r => setTimeout(r, 50));
    }
};

const props = withDefaults(defineProps<{
    visible: boolean;
    title?: string;
    placeholder?: string;
    maxlength?: number;
    defaultValue?: string;
    errorTip?: string;
    cancelText?: string;
    confirmText?: string;
}>(), {
    title: '修改名称',
    placeholder: '请输入名称',
    maxlength: 40,
    defaultValue: '',
    errorTip: '',
    cancelText: '取消',
    confirmText: '完成',
});

// eslint-disable-next-line no-spaced-func
const emit = defineEmits<{
    (e: 'update:visible', val: boolean): void;
    (e: 'close'): void;
    (e: 'confirm', value: string): void;
}>();

const inputValue = ref('');
const inputRef = ref<HTMLInputElement>();
const ready = ref(false);
const showEmojiError = ref(false);

const emojiRegex = /\p{Emoji_Presentation}|\p{Extended_Pictographic}/u;
const hasEmoji = computed(() => emojiRegex.test(inputValue.value));
const canSubmit = computed(() => !!inputValue.value.trim());

// 长度兜底：中文输入法 composition 阶段、历史超长数据回填等场景，
// 原生 maxlength 不一定生效，这里强制截断到 maxlength。
watch(inputValue, (val) => {
    if (val && val.length > props.maxlength) {
        inputValue.value = val.slice(0, props.maxlength);
    }
    if (showEmojiError.value) {
        showEmojiError.value = false;
    }
});

watch(() => props.visible, async (val) => {
    if (val) {
        ready.value = false;
        inputValue.value = (props.defaultValue || '').slice(0, props.maxlength);
        // 锁定背景滚动
        document.body.style.overflow = 'hidden';
        await nextTick();
        // 聚焦触发键盘弹出
        inputRef.value?.focus();
        // 等键盘弹出后拿高度
        await fetchKeyboardHeight();
        // 强制把被顶起的页面拉回原位
        if (!isIos) {
            window.scrollTo(0, 0);
            document.body.scrollTop = 0;
            document.documentElement.scrollTop = 0;
        }

        ready.value = true;
    } else {
        ready.value = false;
        kbHeight.value = 0;
        document.body.style.overflow = '';
    }
});

const handleInputFocus = async () => {
    if (kbHeightCache.value) {
        kbHeight.value = kbHeightCache.value;
    } else {
        await fetchKeyboardHeight();
    }
    if (!isIos) {
        window.scrollTo(0, 0);
        document.body.scrollTop = 0;
        document.documentElement.scrollTop = 0;
    }
};

const handleClose = () => {
    emit('update:visible', false);
    emit('close');
};

const handleConfirm = () => {
    if (!canSubmit.value) return;
    if (hasEmoji.value) {
        showEmojiError.value = true;
        return;
    }
    emit('confirm', inputValue.value.trim());
};

const handleAddToAlbum = async (height: number) => {
    if (height) {
        kbHeightCache.value = height;
    }
    kbHeight.value = height || 0;
    if (!height) {
        await nextTick();
        inputRef.value?.blur();
    }
};

onMounted(async () => {
    registerSetKeyboardHeight(handleAddToAlbum);
});
</script>
