 ```typescript
 // CodeAtomicMarkdown.vue:70-87
 // 核心设计：区分"用户真实操作"和"程序回响"两类 scroll

 // scroll 事件（含程序回响、hljs 布局重排）→ 只允许"贴底时单向翻 true"

     if (isNearBottom(scrollEl)) autoFollow = true;  // 只单向

     userIntentRafId = window.requestAnimationFrame(() => {
         autoFollow = isNearBottom(scrollEl); // 双向
     });

 // MutationObserver 监听内容变化 → 触发 scrollToBottom

 contentObserver.observe(el, { childList: true, subtree: true, characterData: true });
 ```

 ```typescript
 // CodeAtomicMarkdown.vue:169-181
 // 难点：atomic-markdown 内部会重建 DOM，需要重新查找 scrollEl

     if (scrollEl && codeContentWrapperRef.value?.contains(scrollEl)) return;
 ```
